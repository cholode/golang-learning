

### overlaps 源码

```go

func overlaps[E any](a, b []E) bool {//判断两个切片是否共享了部分内存
	if len(a) == 0 || len(b) == 0 {//如果存在切片长度为0
		return false
	}
	elemSize := unsafe.Sizeof(a[0])//计算单个元素大小，any大小不固定
	if elemSize == 0 {//是struct{}直接返回
		return false
	}
	//通过占用范围判断是否有重叠部分
	return uintptr(unsafe.Pointer(&a[0])) <= uintptr(unsafe.Pointer(&b[len(b)-1]))+(elemSize-1) &&
		uintptr(unsafe.Pointer(&b[0])) <= uintptr(unsafe.Pointer(&a[len(a)-1]))+(elemSize-1)
}


```

### rotateRight 源码

```go
func rotateLeft[E any](s []E, r int) {
	Reverse(s[:r])
	Reverse(s[r:])
	Reverse(s)
}
```
**第一步：翻转前 r 个元素**

```go
Reverse(s[:r]) // 翻转前 2 个
```
执行前：[1, 2, 3, 4, 5, 6]
执行后：[2, 1, 3, 4, 5, 6]
第二步：翻转 r 之后的剩余元素

```go
Reverse(s[r:]) // 翻转索引 2 到末尾的所有元素
```
执行前：[2, 1, 3, 4, 5, 6]
执行后：[2, 1, 6, 5, 4, 3]
第三步：翻转整个切片

```go
Reverse(s) // 翻转全部元素
```
执行前：[2, 1, 6, 5, 4, 3]
执行后：[3, 4, 5, 6, 1, 2]
向右平移r位


### Insert 源码
```go
func Insert[S ~[]E, E any](s S, i int, v ...E) S {
	_ = s[i:] // bounds check

	m := len(v)
	if m == 0 {
		return s
	}
	n := len(s)
	if i == n {//如果是接在末尾
		return append(s, v...)
	}

	if n+m > cap(s) {//如果超过容量
		s2 := append(s[:i], make(S, n+m-i)...)//直接开新的容量
		copy(s2[i:], v)//覆盖后面的v
		copy(s2[i+m:], s[i:])//然后再把末尾盖上去
		return s2//返回新的切片
	}
	s = s[:n+m]//扩大长度


	if !overlaps(v, s[i+m:]) {//如果没有重复

		copy(s[i+m:], s[i:])//先把被挤的复制到后面，如果重叠的话会导致插入切片被这一步改变

		copy(s[i:], v)//再把要插入了复制到中间

		return s
	}

	copy(s[n:], v)//复制到不会受影响的右边

	rotateRight(s[i:], m)//平移m位
	return s
}
```

### grow 源码

```go
func Grow[S ~[]E, E any](s S, n int) S {// 剩余容量设置为至少为n
    // 负数容量直接 panic，和内置 append 行为对齐
	if n < 0 {
		panic("cannot be negative")
	}
    // 计算「真正需要新增的容量」
    // cap(s)-len(s) 是当前切片已经有的空闲容量
    // 用 n 减去空闲容量：如果结果 ≤ 0，说明现有容量足够，不需要扩容
	if n -= cap(s) - len(s); n > 0 {
        // s[:cap(s)]长度拉到容量
        // make([]E, n)...增加n的长度
        // [:len(s)]截取长度为len(s)的部分，容量不会缩短
		s = append(s[:cap(s)], make([]E, n)...)[:len(s)]
	}
	return s
}
```

### Delete 源码

```go
// Delete 从切片 s 中移除索引区间 [i, j) 内的所有元素，返回移除元素后的新切片。

//   1. 索引越界时会触发运行时 panic，行为与原生切片索引逻辑一致
//   2. 当 i == j 时走快速路径，直接返回原切片，无额外内存开销
//   3. 自动清零尾部废弃元素，解除指针引用，辅助垃圾回收，避免内存滞留
func Delete[S ~[]E, E any](s S, i, j int) S {
	// 利用三索引切片表达式触发 Go 原生边界检查
	// 若 i、j 超出合法范围（i<0 / j>len(s) / i>j），此处会直接抛出越界 panic
	// 无需手动编写条件判断，是 Go 标准库的惯用高效写法
	_ = s[i:j:len(s)] // bounds check

	// 无元素需要删除的快速路径，直接返回原切片
	if i == j {
		return s
	}

	oldlen := len(s)                 // 记录切片原始长度，用于定位尾部的废弃元素区间
	s = append(s[:i], s[j:]...)      // 拼接前半段 [0:i) 和后半段 [j:)，实现移除中间 [i,j) 区间元素
	clear(s[len(s):oldlen])          // 将尾部已失效的元素置为对应类型的零值
	                                 // 若元素为指针/引用类型，可解除对底层对象的引用，帮助 GC 及时回收，避免内存泄漏
	return s
}
```

### Replace 源码

```go

// Replace 将切片 s 中索引区间 [i, j) 内的元素替换为单个元素或切片 v 中的元素，返回替换后的新切片。

// 核心特性：
//   1. 索引越界时触发原生运行时 panic，行为与切片索引完全一致
//   2. 优先原地修改，仅容量不足时才分配新内存，性能最优
//   3. 自动清零尾部废弃元素，解除引用以辅助 GC 回收，避免内存滞留
//   4. 完整处理 v 与源切片内存重叠的场景（如 v 是 s 的子切片），保证拷贝正确性
func Replace[S ~[]E, E any](s S, i, j int, v ...E) S {
	_ = s[i:j] // bounds check

	// 替换区间长度为 0，等价于在位置 i 插入元素 v，直接复用 Insert 函数逻辑
	if i == j {
		return Insert(s, i, v...)
	}

	// 替换区间恰好到达切片末尾，无需处理后半段元素的前移
	if j == len(s) {
		// 直接拼接前半段 s[:i] 和替换元素 v，得到结果切片
		s2 := append(s[:i], v...)
		// 若替换后长度小于原长度，说明底层数组尾部存在废弃元素
		// 清零该区域以解除对象引用，帮助 GC 及时回收，避免隐性内存占用
		if len(s2) < len(s) {
			clear(s[len(s2):]) // zero/nil out the obsolete elements, for GC
		}
		return s2
	}

	// tot：替换完成后切片的最终总长度 = 前半段长度 + 替换元素长度 + 后半段长度
	tot := len(s[:i]) + len(v) + len(s[j:])
	// 最终长度超过原切片容量，无法原地容纳，必须分配新的底层数组
	if tot > cap(s) {
		// Too big to fit, allocate and copy over.
		// 一次性分配容量为 tot 的新切片：复用前半段 s[:i] 的元素，后续扩展至总长度
		// 该写法一次性申请足够内存，避免 append 过程中多次扩容拷贝
		s2 := append(s[:i], make(S, tot-i)...) // See Insert
		// 拷贝替换元素 v 到对应位置
		copy(s2[i:], v)
		// 拷贝原切片后半段元素到替换内容之后
		copy(s2[i+len(v):], s[j:])
		return s2
	}

	// 容量充足，可原地完成替换，先构造最终长度的切片视图 r
	r := s[:tot]

	// 场景1：替换元素长度 ≤ 被删除区间长度（替换后总长度不变或缩短）
	// 此时 v 可完全放入被删除的区间内，无内存重叠风险，操作逻辑最简单
	if i+len(v) <= j {
		// Easy, as v fits in the deleted portion.
		// 第一步：将替换元素 v 拷贝到起始位置 i
		copy(r[i:], v)
		// 第二步：将原切片后半段 s[j:] 拷贝到替换元素的末尾
		copy(r[i+len(v):], s[j:])
		// 清零尾部超出最终长度的废弃元素，解除引用辅助 GC
		clear(s[tot:]) // zero/nil out the obsolete elements, for GC
		return r
	}

	// 以下为场景2：替换元素长度 > 被删除区间长度（替换后总长度变长）
	// 核心难点：需处理 v 与源切片内存重叠的问题（例如 v 是 s 的子切片时）
	// 避免拷贝过程中数据被提前覆盖，导致最终结果错误

	// 子场景2.1：后半段拷贝目标区域 与 v 无内存重叠
	// 可先拷贝后半段，再拷贝替换元素，不会出现数据覆盖问题
	if !overlaps(r[i+len(v):], v) {
		// Easy, as v is not clobbered by the first copy.
		// 先拷贝原切片后半段到靠后的最终位置（不会覆盖 v 的数据）
		copy(r[i+len(v):], s[j:])
		// 再拷贝替换元素 v 到起始位置
		copy(r[i:], v)
		return r
	}

	// y：替换后新增的元素长度 = 替换元素总长度 - 被删除区间长度
	// 也就是切片尾部需要向外扩展的元素数量
	y := len(v) - (j - i) // length of y portion

	// 子场景2.2：被删除区间 [i:j] 与 v 无内存重叠
	// 分两段拷贝 v，再通过右旋完成元素位置对齐
	if !overlaps(r[i:j], v) {
		// 将 v 的后段（y 位置之后的部分）拷贝到原删除区间
		copy(r[i:j], v[y:])
		// 将 v 的前段（前 y 个元素）拷贝到切片尾部扩展区域
		copy(r[len(s):], v[:y])
		// 将 [i:] 区间的元素右旋 y 位，完成最终位置对齐
		rotateRight(r[i:], y)
		return r
	}
	// 子场景2.3：尾部扩展区域 与 v 无内存重叠
	if !overlaps(r[len(s):], v) {
		// 先将 v 的前段拷贝到尾部扩展区
		copy(r[len(s):], v[:y])
		// 再将 v 的后段拷贝到原删除区间
		copy(r[i:j], v[y:])
		// 右旋对齐元素位置
		rotateRight(r[i:], y)
		return r
	}

	// 子场景2.4：v 与源切片完全重叠的极端场景
	// 通过计算重叠起始偏移量 k，分两段拷贝避免数据覆盖
	k := startIdx(v, s[j:])
	// 先完整拷贝 v 到起始位置
	copy(r[i:], v)
	// 拷贝剩余未重叠的后半段元素，完成最终拼接
	copy(r[i+len(v):], r[i+k:])
	return r
}
```
